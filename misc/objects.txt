		Человек
			высотаа = 7-8пикс
			Ширина = 4пикс
			
		Окно жилого дома, высота = 4пикс
		
		Дверной проём
			высота = 9пикс (или 8рикс если это вдерь в помещение с низкими потолками, например в дом-трейлер)
			ширина = 4пикс
			
		Этаж, высота = 10-11пикс. (но в доме-трейлере это было бы 9пикс)
		
		Макс высота здания = 80пикс
